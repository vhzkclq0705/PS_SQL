```sql
-- Lv.5
-- 멸종위기의 대장균 찾기

WITH RECURSIVE TEMP AS (
    SELECT ID, PARENT_ID, 1 AS GEN
    FROM ECOLI_DATA
    WHERE PARENT_ID IS NULL
    
    UNION ALL
    
    SELECT C.ID, C.PARENT_ID, P.GEN + 1 AS GEN
    FROM ECOLI_DATA AS C
    JOIN TEMP AS P
    ON C.PARENT_ID = P.ID
),
GEN_INFO AS (
    SELECT ID, GEN
    FROM TEMP
    WHERE ID NOT IN (
        SELECT PARENT_ID
        FROM TEMP
        WHERE PARENT_ID IS NOT NULL
    )
)

SELECT COUNT(*) AS COUNT, GEN AS GENERATION
FROM GEN_INFO
GROUP BY GEN
ORDER BY GEN
```
