```sql
SELECT
    info.rest_id,
    info.rest_name,
    info.food_type,
    info.favorites,
    info.address,
    ROUND(AVG(review.review_score), 2) AS score
FROM rest_info info
INNER JOIN rest_review review
ON info.rest_id = review.rest_id
WHERE SUBSTR(address, 1, 2) = '서울'
GROUP BY 
    info.rest_id,
    info.rest_name,
    info.food_type,
    info.favorites,
    info.address
ORDER BY
    score DESC,
    info.favorites DESC
```
