```sql
-- Lv.2
-- 조건에 맞는 사원 정보 조회하기

WITH sum_score AS (
    SELECT 
        emp_no,
        SUM(score) AS score
    FROM hr_grade
    WHERE year = 2022
    GROUP BY emp_no
)

SELECT
    a.score,
    a.emp_no,
    b.emp_name,
    b.position,
    b.email
FROM sum_score a
JOIN hr_employees b ON a.emp_no = b.emp_no
WHERE a.score = (
    SELECT MAX(score)
    FROM sum_score
)
```
