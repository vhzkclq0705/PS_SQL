```sql
-- Lv.3
-- 즐겨찾기가 가장 많은 식당 정보 출력하기

WITH ranked AS (
    SELECT
        food_type,
        rest_id,
        rest_name,
        favorites,
        RANK() OVER (PARTITION BY food_type ORDER BY favorites DESC) AS rn
    FROM REST_INFO
)

SELECT
    food_type,
    rest_id,
    rest_name,
    favorites
FROM ranked
WHERE rn = 1
ORDER BY food_type DESC;
```
