```sql
-- Lv.4
-- 입양 시각 구하기(2)

WITH RECURSIVE hours AS (
    SELECT 0 AS hour
    
    UNION ALL
    
    SELECT hour + 1
    FROM hours
    WHERE hour < 23
)

SELECT
    a.hour,
    COALESCE(b.count, 0) AS count
FROM hours a
LEFT JOIN (
    SELECT
        HOUR(datetime) AS hour,
        COUNT(*) AS count
    FROM animal_outs
    GROUP BY hour
) b ON a.hour = b.hour
ORDER BY a.hour;
```
