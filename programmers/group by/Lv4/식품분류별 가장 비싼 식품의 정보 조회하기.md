```sql
-- Lv.4
-- 식품분류별 가장 비싼 식품의 정보 조회하기

WITH ranked AS (
    SELECT
        category,
        product_name,
        price,
        RANK() OVER (PARTITION BY category ORDER BY price DESC) AS rn
    FROM food_product
)

SELECT
    category,
    price AS max_price,
    product_name
FROM ranked
WHERE category IN ('과자', '국', '김치', '식용유') AND rn = 1
ORDER BY price DESC;
```
