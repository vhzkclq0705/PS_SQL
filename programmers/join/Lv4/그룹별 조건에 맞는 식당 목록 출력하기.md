-- Lv.4
-- 그룹별 조건에 맞는 식당 목록 출력하기

-- CTE
WITH REVIEW_CNT AS (
    SELECT MEMBER_ID, COUNT(*) AS CNT
    FROM REST_REVIEW
    GROUP BY MEMBER_ID
),
TOP_REVIEWER AS (
    SELECT MEMBER_ID
    FROM REVIEW_CNT
    WHERE CNT = (
        SELECT MAX(CNT)
        FROM REVIEW_CNT
    )
)
SELECT
    A.MEMBER_NAME,
    B.REVIEW_TEXT,
    DATE_FORMAT(B.REVIEW_DATE, '%Y-%m-%d') AS REVIEW_DATE
FROM MEMBER_PROFILE A
JOIN REST_REVIEW B ON A.MEMBER_ID = B.MEMBER_ID
JOIN TOP_REVIEWER C ON A.MEMBER_ID = C.MEMBER_ID
ORDER BY B.REVIEW_DATE, B.REVIEW_TEXT

-- 서브쿼리
SELECT 
    A.MEMBER_NAME,
    B.REVIEW_TEXT, 
    DATE_FORMAT(B.REVIEW_DATE, '%Y-%m-%d') AS REVIEW_DATE
FROM MEMBER_PROFILE A
JOIN REST_REVIEW B ON A.MEMBER_ID = B.MEMBER_ID
WHERE A.MEMBER_ID IN (
    SELECT MEMBER_ID
    FROM REST_REVIEW
    GROUP BY MEMBER_ID
    HAVING COUNT(*) = (
        SELECT MAX(CNT)
        FROM (
            SELECT COUNT(*) AS CNT
            FROM REST_REVIEW
            GROUP BY MEMBER_ID
        ) as SUB
    )
)
ORDER BY REVIEW_DATE, REVIEW_TEXT
